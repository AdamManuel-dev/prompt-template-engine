```typescript
interface Policy {
  subjects: string[];    // Users, roles, patterns
  resources: string[];   // Resource patterns
  actions: string[];     // Action patterns
  rules: PolicyRule[];   // Evaluation rules
  conditions: PolicyCondition[]; // Context conditions
  effect: 'allow' | 'deny';
}
```

**Example Policy:**
```typescript

  ]
};
```
